Above we see the NAs.
Now we'll replace them with the mean.

impute <- function(x, fun) {
    missing <- is.na(x)
    replace(x, missing, fun(x[!missing]))
}
act <- activity
act$steps <- ifelse(is.na(activity$steps), impute(activity$steps, mean), activity$steps)
